def func_q5(prod,tp):
    
    Q5=pd.DataFrame(cust_repay[cust_repay.Product==prod].groupby(by=[tp,'Customer']).
             Amount.sum().reset_index().groupby(by='Customer').Amount.sum())
    res= Q5.sort_values(by='Amount',ascending=False).iloc[0:10]
    return(res)
