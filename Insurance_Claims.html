# H0: Amount claimed by males is same as amount claimed by females
# H1: Amount claimed by males and females are significantly different

# CI:95%  , pvalue=0.05  (alpha region cutoff value)

tstat,pvalue=stats.ttest_ind(data_male.claim_amount,data_female.claim_amount)

print('t statsitc value is : ',tstat ,'\t  p value is : ',pvalue)

if pvalue>0.05:
    print('We fail to reject the H0 ')
else:
    print('We reject the H0 ')
